JOINS:

SELECT COL1, COL2
FROM TAB1, TAB2
WHERE TAB1.COMM_COL == TAB2.COMM_COL


INNER JOIN:
	SELECT COL1,COL2 FROM TAB1 INNER JOIN TAB2 WHERE TAB1.COMM_COL == TAB2.COMM_COL
LEFT JOIN:
	SELECT COL1, COL2 FROM TAB1 LEFT JOIN TAB2 WHERE TAB1.COMM_COL == TAB2.COMM_COL
RIGHT JOIN:
	SELECT COL1, COL2 FROM TAB1 RIGH JOIN TAB2 WHERE TAB1.COMM_COL == TAB2.COMM_COL
FULL JOIN:
	SELECT COL1, COL2 FROM TAB1 FULL JOIN TAB2 WHERE TAB1.COMM_COL == TAB2.COMM_COL

IF YOUR DATABASE DOESN'T SUPPORT FULL JOIN YOU CAN USE "UNION ALL"

	SELECT COL1,COL2 FROM TAB1 LEFT JOIN TAB2 WHERE TAB1.COMM_COL == TAB2.COMM_COL
	UNION ALL
	SELECT COL1,COL2 FROM TAB1 RIGHT JOIN TAB2 WHERE TAB1.COMM_COL == TAB2.COMM_COL
